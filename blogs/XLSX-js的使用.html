    XLSX.js的使用

    GaoChuang 7/30/2020 XLSXjs

    SheetJS js-xlsx (opens new window) 中文文档： https://github.com/rockboom/SheetJS-docs-zh-CN (opens new window)

    # 使用步骤

    // 从头开始创建工作簿
var wb = XLSX.utils.book_new();
/**
 * 创建工作表
 * aoa_to_sheet    二维数组
 * json_to_sheet   对象数组
 * table_to_sheet  tableDOM
 */
let ws = XLSX.utils.json_to_sheet(json);
// 把工作表添加到工作簿中
XLSX.utils.book_append_sheet(wb, ws, "sheet");
// 写入 （node）
XLSX.writeFile(wb, path.resolve(__dirname, "./test.xlsx"), {
  type: "buffer",
  Props: { Author: "author" },
});
